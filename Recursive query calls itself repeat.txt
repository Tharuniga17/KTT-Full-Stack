Recursive query calls itself repeatedly until certain condition is met 

every query will have 3 parts

with recursive name as (
     ---starting point---
     union all 
     ---recursive case---
)
select * from name;

base --> starting value , recursive --> repeat logic , Condition --> when to stop

with recursive numbers as (
select 1 as n -- base 
union all 
select n + 1 from numbers where n < 5
)
select * from numbers;

select * from (select name, subject, am from student_marks) as source pivot (max(marks) for subject in ([Math],[Science]) as pivot_table;

unpivot --> will convert columns to rows

select name ,subject, marks from academics unpivot(marks for subject in (Math,Science) ) as unpivot_table;
